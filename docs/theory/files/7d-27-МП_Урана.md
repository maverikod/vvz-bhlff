def legendre_P2(c):
    return 0.5*(3*c*c - 1.0)

def dP2_dtheta(theta):
    return -3.0*np.sin(theta)*np.cos(theta)

def sph_components_from_axis(rvec, n):
    r = np.linalg.norm(rvec)
    if r == 0:
        return 0.0, 1.0, 0.0, np.array([0,0,0]), np.array([0,0,0])
    rhat = rvec / r
    c = np.dot(rhat, n)
    s = np.sqrt(max(0.0, 1.0 - c*c))
    n_perp = n - c*rhat
    n_perp_norm = np.linalg.norm(n_perp)
    if n_perp_norm < 1e-12:
        if abs(rhat[2]) > 0.9:
            etheta = np.array([1.0, 0.0, -rhat[0]/max(1e-12, rhat[2])])
        else:
            etheta = np.array([-rhat[1], rhat[0], 0.0])
        etheta = etheta/np.linalg.norm(etheta)
    else:
        etheta = -n_perp / n_perp_norm
    return r, c, s, rhat, etheta

def B_multipole_outside(rvec):
    r, c, s, er, etheta = sph_components_from_axis(rvec, n_axis)
    if r < 1e-9:
        return np.zeros(3)
    Br1 = 2.0*A1*c / (r**3)
    Bth1 = A1*s / (r**3)
    P2 = legendre_P2(c)
    dP2 = dP2_dtheta(np.arccos(np.clip(c, -1.0, 1.0)))
    Br2 = 3.0*A2*P2 / (r**4)
    Bth2 = A2*dP2 / (r**4)
    Br = Br1 + Br2
    Bth = Bth1 + Bth2
    B = Br*er + Bth*etheta
    return B

def B_total(rvec):
    r = np.linalg.norm(rvec)
    if r <= rc:
        return B_core_in * n_axis + Bsun_vec
    return B_multipole_outside(rvec) + Bsun_vec

def rotation_matrix_from_z_to_n(n):
    z = np.array([0.0,0.0,1.0])
    v = np.cross(z, n)
    s = np.linalg.norm(v)
    c = np.dot(z, n)
    if s < 1e-12:
        return np.eye(3) if c>0 else -np.eye(3)
    vx = np.array([[0,-v[2],v[1]],[v[2],0,-v[0]],[-v[1],v[0],0]])
    return np.eye(3) + vx + vx@vx*((1-c)/(s**2))

Rm = rotation_matrix_from_z_to_n(n_axis)

    rvec_axis = np.array([np.sin(theta), 0.0, np.cos(theta)])  # unit radius in axis frame
    rglob = Rm @ rvec_axis
    B = B_total(rglob)
    return np.linalg.norm(B)

# Почему «как у Урана» почти не видно на других

1. **Нет нужной оболочки.** У Урана (и частично Нептуна) оболочка сильно неоднородна: $R/T$ по слоям анизотропны → большой вклад $l=2,3$. У Юпитера/Сатурна оболочка ближе к осесимметричной → подавление квадруполя.

2. **Смещение центра и несоосность.** У Урана центр источника сдвинут (\~0.3 R), оси ядра/оболочки/вращения несоосны → чётные гармоники усиливаются. У Земли/Юпитера смещения малы.

3. **Большой наклон оси.** «Лежка» сама квадруполь не создаёт, но делает его **видимым** (солнце–планета) и даёт яркую сетку нулей. У Земли наклон \~11°, у Сатурна \~0° — эффекта почти не видно.

4. **Перколяция и потери.** У Урана перколяция наружу $\Pi_{\rm out}$ мала локально, потери $\chi''$ невелики → ячейки устойчивы. У других — либо каналы открыты, либо потери сглаживают структуру.

5. **Магнитопауза и драйв.** Размер/формa МП и ориентировка IMF: у Урана сочетание параметров «подсвечивает» ячейки. У Юпитера мощный почти диполь «перешибает» наложение; у Земли сетка есть, но слабее и «замыта» симметрией.

**Итого:** нужна тройка условий — *анизотропная расслаивающаяся оболочка* + *смещённый/несоосный источник* + *большой наклон*. Это выполняется у Урана (и близко у Нептуна), но не выполняется у Земли/Юпитера/Сатурна.
